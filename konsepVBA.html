Bab 01
Apa itu VBA?
Pengenalan Visual Basic for Applications

VBA (Visual Basic for Applications) adalah bahasa pemrograman bawaan Microsoft Office yang memungkinkan pengguna untuk mengotomasi tugas-tugas berulang, membuat fungsi kustom, dan membangun aplikasi berbasis Excel tanpa memerlukan perangkat lunak eksternal.

⚡
Otomasi
Jalankan ratusan tugas berulang dalam hitungan detik dengan satu klik.
📐
Fungsi Kustom
Buat rumus Excel sendiri (UDF) sesuai kebutuhan kalkulasi Anda.
🖼️
Antarmuka
Desain form dialog interaktif dengan tombol, kotak teks, dan daftar pilihan.
🔗
Integrasi Office
Kontrol Word, PowerPoint, dan Outlook langsung dari Excel via VBA.

Sejarah Singkat

VBA diperkenalkan oleh Microsoft pada tahun 1993 sebagai pengganti macro bahasa Excel yang lama. Berbasis pada Visual Basic 6.0, VBA menjadi standar otomasi di seluruh produk Microsoft Office. Hingga kini, VBA tetap menjadi alat yang paling banyak digunakan untuk kustomisasi Excel di lingkungan bisnis dan akademik.

💡
Mengapa belajar VBA? Dalam dunia pendidikan matematika, VBA memungkinkan pembuatan alat evaluasi otomatis, generator soal, analisis data mahasiswa, dan simulasi matematika langsung di Excel yang sudah familiar bagi semua orang.

Bab 03
Variabel & Tipe Data
Menyimpan data dalam program VBA

Variabel adalah tempat penyimpanan sementara dalam memori. Di VBA, variabel dideklarasikan dengan kata kunci Dim dan wajib diberi tipe data (praktik terbaik).

Deklarasi Variabel

variabel.bas 
Sub ContohVariabel()
    ' ── Deklarasi dengan Dim ──
    Dim namaMahasiswa  As String
    Dim nim            As Long
    Dim nilaiUjian     As Double
    Dim lulus          As Boolean
    Dim tanggalLahir   As Date
    
    ' ── Pengisian nilai ──
    namaMahasiswa = "Budi Santoso"
    nim           = 2024001
    nilaiUjian    = 87.5
    lulus         = True
    tanggalLahir  = #01/15/2002#
    
    ' ── Menampilkan hasil ──
    MsgBox "Nama : " & namaMahasiswa & Chr(10) & _
           "NIM  : " & nim & Chr(10) & _
           "Nilai: " & nilaiUjian & Chr(10) & _
           "Lulus: " & lulus
End Sub

Bab 04
Operator VBA
Operasi aritmatika, perbandingan, dan logika
operator.bas 
Sub ContohOperator()
    Dim a As Integer, b As Integer
    a = 17 : b = 5
    
    ' ── ARITMATIKA ──
    Debug.Print a + b      ' 22  — Penjumlahan
    Debug.Print a - b      ' 12  — Pengurangan
    Debug.Print a * b      ' 85  — Perkalian
    Debug.Print a / b      ' 3.4 — Pembagian
    Debug.Print a \ b      ' 3   — Pembagian bulat (integer div)
    Debug.Print a Mod b   ' 2   — Sisa bagi (modulo)
    Debug.Print a ^ b      ' 1419857 — Perpangkatan
    
    ' ── PERBANDINGAN (menghasilkan True/False) ──
    Debug.Print a = b       ' False — sama dengan
    Debug.Print a <> b      ' True  — tidak sama
    Debug.Print a > b       ' True  — lebih besar
    Debug.Print a <= b      ' False — kurang dari atau sama
    
    ' ── LOGIKA ──
    Debug.Print (a > 10) And (b < 10)  ' True
    Debug.Print (a > 20) Or  (b < 10)  ' True
    Debug.Print Not (a = b)               ' True
    
    ' ── PENGGABUNGAN STRING ──
    Debug.Print "Nilai a = " & a  ' "Nilai a = 17"
End Sub

Sub ContohMsgBox()

    End Select

' ── InputBox: meminta input dari pengguna ──
Sub ContohInputBox()
    Dim nama   As String
    Dim nilai  As Double
    
    ' InputBox(prompt, title, defaultValue)
    nama  = InputBox("Masukkan nama mahasiswa:", "Input Nama", "Nama Lengkap")
    nilai = CDbl(InputBox("Masukkan nilai (0-100):", "Input Nilai", "0"))
    
    ' Validasi — cek apakah dibatalkan
    If nama = "" Then
        MsgBox "Input dibatalkan.", vbInformation
        Exit Sub
    End If
    
    ' Proses dan tampilkan
    Dim grade As String
    If     nilai >= 85 Then : grade = "A"
    ElseIf nilai >= 75 Then : grade = "B"
    ElseIf nilai >= 65 Then : grade = "C"
    Else                   : grade = "D"
    End If
    
    MsgBox nama & " → Nilai: " & nilai & " → Grade: " & grade
End Sub

Sub ForNextDemo()

    Dim total As Long
    
    ' Isi nilai ke kolom A, hitung total

' ══════════ FOR - STEP (mundur) ══════════
Sub ForStepMundur()
    Dim i As Integer
    For i = 10 To 1 Step -1
        Debug.Print i   ' 10, 9, 8, ..., 1
    Next i
End Sub

' ══════════ DO WHILE ══════════
Sub DoWhileDemo()
    Dim baris As Integer
    baris = 2
    
    ' Baca data sampai sel kosong
    Do While Cells(baris, 1).Value <> ""
        Cells(baris, 3).Value = Cells(baris, 2).Value * 1.1
        baris = baris + 1
    Loop
End Sub

' ══════════ FOR EACH (iterasi koleksi) ══════════
Sub ForEachDemo()
    Dim ws As Worksheet
    
    ' Iterasi semua sheet dalam workbook
    For Each ws In ThisWorkbook.Worksheets
        Debug.Print ws.Name
    Next ws
    
    ' Iterasi semua sel dalam range

        If cel.Value > 50 Then
            cel.Interior.Color = RGB(198, 239, 206)  ' Hijau muda
        End If
    Next cel
End Sub

Function HitungPPN(harga As Double, persen As Double) As Double
    HitungPPN = harga * (persen / 100)
End Function

' Grade dari nilai numerik
Function TentukanGrade(nilai As Double) As String
    Select Case nilai
        Case Is >= 85 : TentukanGrade = "A"
        Case Is >= 75 : TentukanGrade = "B"
        Case Is >= 65 : TentukanGrade = "C"
        Case Is >= 55 : TentukanGrade = "D"
        Case Else     : TentukanGrade = "E"
    End Select
End Function

' Faktorial rekursif
Function Faktorial(n As Long) As Long
    If n <= 1 Then
        Faktorial = 1
    Else
        Faktorial = n * Faktorial(n - 1)
    End If
End Function

Sub ManipulasiRange()
    ' ── Berbagai cara merujuk sel ──
    Range("A1").Value         = "Satu sel"
    Range("A1:C5").Value      = "Range"    ' Isi semua
    Cells(1, 1).Value          = "Cells(baris,kol)"
    Range(Cells(1,1), Cells(5,3)).Value = 0
    
    ' ── Properti penting ──
    Debug.Print Range("A1:C5").Rows.Count     ' 5
    Debug.Print Range("A1:C5").Columns.Count  ' 3
    Debug.Print Range("A1:C5").Address        ' $A$1:$C$5
    
    ' ── Baris terakhir berisi data ──
    Dim lastRow As Long
    lastRow = Cells(Rows.Count, 1).End(xlUp).Row
    MsgBox "Baris data terakhir: " & lastRow
    
    ' ── Format otomatis ──
    With Range("A1:E" & lastRow)
        .Borders.LineStyle     = xlContinuous
        .Font.Name              = "Calibri"
        .Columns.AutoFit
    End With
    
    ' ── Conditional formatting via VBA ──
    Dim cel As Range
    For Each cel In Range("B2:B" & lastRow)
        Select Case cel.Value
            Case Is >= 85 : cel.Interior.Color = RGB(198,239,206) ' Hijau
            Case Is >= 70 : cel.Interior.Color = RGB(255,242,204) ' Kuning
            Case Else     : cel.Interior.Color = RGB(255,199,206) ' Merah
        End Select
    Next cel
End Sub

' ── Referensi Worksheet ──
Sub KelolaSheet()
    Dim ws   As Worksheet
    Dim wsNw As Worksheet
    
    ' Cara merujuk sheet
    Set ws = Worksheets("Sheet1")         ' Nama
    Set ws = Worksheets(1)               ' Index
    Set ws = ActiveSheet                   ' Sheet aktif
    
    ' Tambah sheet baru
    Set wsNw = Worksheets.Add()
    wsNw.Name = "Data Mahasiswa"
    
    ' Salin, hapus, sembunyikan
    ws.Copy After:=Worksheets(Worksheets.Count)
    ' ws.Delete   — hapus sheet
    ws.Visible = xlSheetHidden   ' sembunyikan
End Sub

' ── Kelola Workbook ──
Sub KelolaWorkbook()
    Dim wb As Workbook
    
    ' Buka file yang ada
    Set wb = Workbooks.Open("C:\Data\Nilai.xlsx")
    
    ' Simpan dengan nama baru

        FileFormat:=xlOpenXMLWorkbookMacroEnabled
    
    ' Tutup tanpa simpan
    wb.Close SaveChanges:=False
End Sub

Sub ContohArray()
    ' ── Array statis ──
    Dim nilai(1 To 5) As Integer
    nilai(1) = 85
    nilai(2) = 90
    nilai(3) = 78
    nilai(4) = 92
    nilai(5) = 88
    
    ' Hitung rata-rata
    Dim total As Integer, i As Integer
    For i = 1 To 5
        total = total + nilai(i)
    Next i
    MsgBox "Rata-rata: " & (total / 5)
    
    ' ── Array dinamis (ukuran berubah) ──
    Dim dataNama() As String
    Dim jumlah    As Integer
    
    jumlah = CInt(InputBox("Berapa mahasiswa?"))
    ReDim dataNama(1 To jumlah)
    
    For i = 1 To jumlah
        dataNama(i) = InputBox("Nama mahasiswa ke-" & i & ":")
    Next i
    
    ' ── Array 2 dimensi (seperti tabel) ──
    Dim tabel(1 To 3, 1 To 4) As Variant
    ' tabel(baris, kolom) — 3 baris, 4 kolom
    For i = 1 To 3
        Dim j As Integer
        For j = 1 To 4
            tabel(i, j) = i * j   ' tabel perkalian
            Cells(i, j).Value = tabel(i, j)
        Next j
    Next i
End Sub

' ── Kode di balik UserForm Input Nilai ──
' UserForm memiliki:
'   - txtNama       (TextBox) — input nama
'   - txtNilai      (TextBox) — input nilai  
'   - cboProdi      (ComboBox) — pilih prodi
'   - btnSimpan     (CommandButton) — simpan
'   - btnBatal      (CommandButton) — batal

Private Sub UserForm_Initialize()
    ' Isi ComboBox saat form dibuka
    With cboProdi
        .AddItem "Pendidikan Matematika"
        .AddItem "Pendidikan Bahasa Indonesia"
        .AddItem "PGSD"
        .ListIndex = 0   ' Pilih item pertama
    End With
End Sub

Private Sub btnSimpan_Click()
    ' Validasi input
    If txtNama.Text = "" Then
        MsgBox "Nama tidak boleh kosong!", vbExclamation
        txtNama.SetFocus
        Exit Sub
    End If
    
    Dim nilaiInput As Double
    If Not IsNumeric(txtNilai.Text) Then
        MsgBox "Nilai harus berupa angka!", vbExclamation
        txtNilai.SetFocus
        Exit Sub
    End If
    nilaiInput = CDbl(txtNilai.Text)
    
    ' Tulis ke sheet
    Dim ws As Worksheet
    Set ws = Worksheets("Data")
    Dim barisBaru As Long
    barisBaru = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row + 1
    
    ws.Cells(barisBaru, 1) = txtNama.Text
    ws.Cells(barisBaru, 2) = cboProdi.Value
    ws.Cells(barisBaru, 3) = nilaiInput
    
    MsgBox "Data berhasil disimpan!", vbInformation
    txtNama.Text  = ""
    txtNilai.Text = ""
    txtNama.SetFocus
End Sub

Private Sub btnBatal_Click()
    Unload Me   ' Tutup form
End Sub

Kuis 03 — Sub vs Function
Manakah pernyataan yang BENAR tentang perbedaan Sub dan Function di VBA?
ASub bisa digunakan sebagai formula di sel Excel
BFunction mengembalikan nilai; Sub tidak mengembalikan nilai
CSub dan Function adalah hal yang sama
DFunction tidak bisa menerima parameter
Kuis 04 — Objek Excel
Kode VBA mana yang benar untuk mendapatkan baris terakhir yang berisi data di kolom A?

DLastRow = Range("A").Count
Kuis 05 — Loop
Jenis loop mana yang paling tepat untuk mengiterasi semua Worksheet dalam sebuah Workbook?
ADo While dengan counter manual
BFor Each ws In Worksheets
CFor i = 1 To 100
DDo Until selalu lebih efisien
